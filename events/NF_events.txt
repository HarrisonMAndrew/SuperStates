add_namespace = NFEvents

country_event = {
	id = NFEvents.1
	title = "Alliance Offer"
	desc = "We have been invited to an alliance with [BAN.GetName]. Should we accept the invitation?"
	picture = news_event_generic_sign_treaty1

	is_triggered_only = yes

	option = {
		name = "Yes"
		ai_chance = {
			factor = 3
			modifier = {
				factor = 0
				NOT = { has_government = BAN }
			}
			modifier = {
				factor = 0.5
				has_opinion = {
					target = BAN
					value < 100
				}
			}
			modifier = {
				factor = 0.5
				has_opinion = {
					target = BAN
					value < 50
				}
			}
			modifier = {
				factor = 0
				has_opinion = {
					target = BAN
					value < 0
				}
			}
		}

		if = {
			limit = {
				BAN = {
					is_faction_leader = no
				}
			}
			BAN = {
				leave_faction = yes
				create_faction = "league_of_the_fifth_international"
			}
		}
		BAN = {
			add_to_faction = ROOT
		}
		hidden_effect = {
			news_event = { id = news.502 }
		}
	}

	option = {
		name = "No"
		ai_chance = { factor = 1 }
		hidden_effect = {
			news_event = { id = news.503 }
		}
	}
}

country_event = {
	id = NFEvents.2
	title = "[USA.GetName] Demands [CSA.GetName]!"
	desc = "Today [USA.GetName] sent an envoy to us with a proposition of an union. We are alone and in this world, and a union with [USA.GetName might prove to be fruitful. Our people would be safe with the mighty army of [USA.GetName] and we could possibly flourish with their established economy. Or we could refuse the union which would surely lead to war, but maybe we can hold them off!"
	picture = GFX_report_event_hitler_parade

	is_triggered_only = yes

	option = {
		name = "We accept the Union"
		ai_chance = {
			base = 30
			modifier = {
				factor = 0
				CSA = { is_subject = yes }
			}
			modifier = {
				factor = 0
				NOT = { tag = CSA }
				USA = { on_same_continent = no }
			}
			modifier = {
				add = -15
				USA = { has_army_size = { size < 40 } }
			}
			modifier = {
				add = 45
				USA = { has_army_size = { size > 39 } }
			}
		}
		FROM = {
			country_event = { hours = 2 id = NFEvents.3 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}

	option = {
		name = "We Refuse!"
		ai_chance = {
			base = 10
			modifier = {
				factor = 0
				USA = { has_army_size = { size > 39 } }
			}
			modifier = {
				add = 20
				USA = { has_army_size = { size < 30 } }
			}
		}
		FROM = {
			country_event = { hours = 2 id = NFEvents.4 }
			if = {
				limit = { is_in_faction_with = CSA }
				remove_from_faction = CSA
			}
		}
	}
}

country_event = {
	id = NFEvents.4
	title = "[FROM.GetName] Refuses!"
	desc = "[FROM.GetName] Refused our proposed union! This is an insult to us that cannot go unanswered!"
	picture = GFX_report_event_german_troops

	is_triggered_only = yes

	option = {
		name = "It's time for war"
		create_wargoal = {
			type = annex_everything
			target = FROM
		}
		news_event = { id = static_news.3 hours = 6 }
	}
}

country_event = {
	id = NFEvents.3
	title = "[FROM.GetName] accepts!"
	desc = "[FROM.GetName] accepted our proposed union, their added strength will push us to greatness!"
	picture = GFX_report_event_german_speech

	is_triggered_only = yes

	option = {
		name = "A stronger Union!"
		every_country = {
			limit = { is_subject_of = CSA }
			country_event = NFEvents.2
		}
		FROM = {
			every_owned_state = {
				add_core_of = USA
			}
		}
		if = {
			limit = { FROM = { tag = CSA } }
			annex_country = { target = CSA transfer_troops = yes }
			add_political_power = 50
			add_named_threat = { threat = 2 name = "America annexed America" }
			news_event = { id = static_news.2 hours = 6 }
			set_country_flag = CSA_annexed
		}
		else = {
			puppet = FROM
		}

	}
}
